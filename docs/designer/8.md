#### 2.1.4 PTY 测试辅助类

```python
# tests/pty_tests.py
"""PTY 测试辅助类."""

import pty
import os
import select
import subprocess
import time
import fcntl
import struct
import termios
from typing import Optional


class PTYSession:
    """PTY 会话管理器."""
    
    def __init__(self):
        self.master_fd: Optional[int] = None
        self.slave_fd: Optional[int] = None
        self.process: Optional[subprocess.Popen] = None
        self.output_buffer = ""
        
    def start(self, command: str, cwd: Optional[str] = None, env: Optional[dict] = None):
        """启动 PTY 会话."""
        # 创建伪终端
        self.master_fd, self.slave_fd = pty.openpty()
        
        # 设置非阻塞模式
        fl = fcntl.fcntl(self.master_fd, fcntl.F_GETFL)
        fcntl.fcntl(self.master_fd, fcntl.F_SETFL, fl | os.O_NONBLOCK)
        
        # 准备环境变量
        run_env = os.environ.copy()
        if env:
            run_env.update(env)
        
        # 启动子进程
        self.process = subprocess.Popen(
            command,
            shell=True,
            stdin=self.slave_fd,
            stdout=self.slave_fd,
            stderr=self.slave_fd,
            cwd=cwd,
            env=run_env,
            preexec_fn=os.setsid
        )
        
        # 关闭 slave 端（在父进程中）
        os.close(self.slave_fd)
        self.slave_fd = None
        
        # 等待进程启动
        time.sleep(0.5)
        
    def send(self, data: str):
        """发送数据到 PTY."""
        if self.master_fd is not None:
            os.write(self.master_fd, data.encode())
            time.sleep(0.1)  # 给进程处理时间
            
    def read_available(self, timeout: float = 0.5) -> str:
        """读取可用数据."""
        if self.master_fd is None:
            return ""
            
        output = ""
        start_time = time.time()
        
        while time.time() - start_time < timeout:
            readable, _, _ = select.select([self.master_fd], [], [], 0.1)
            if self.master_fd in readable:
                try:
                    data = os.read(self.master_fd, 4096).decode()
                    output += data
                except (OSError, IOError):
                    break
            else:
                if output:  # 如果有输出且暂时没新数据，返回
                    break
                    
        self.output_buffer += output
        return output
        
    def read_until(self, marker: str = "", timeout: float = 5.0) -> str:
        """读取直到看到特定标记或超时."""
        start_time = time.time()
        collected = ""
        
        while time.time() - start_time < timeout:
            chunk = self.read_available(timeout=0.5)
            collected += chunk
            
            if marker and marker in collected:
                return collected
                
            # 检查进程是否还在运行
            if self.process and self.process.poll() is not None:
                break
                
        return collected
        
    def read_all(self, timeout: float = 2.0) -> str:
        """读取所有输出."""
        return self.read_until(timeout=timeout)
        
    def is_terminated(self) -> bool:
        """检查进程是否已终止."""
        return self.process is not None and self.process.poll() is not None
        
    def stop(self):
        """停止 PTY 会话."""
        if self.process and self.process.poll() is None:
            self.process.terminate()
            try:
                self.process.wait(timeout=2)
            except subprocess.TimeoutExpired:
                self.process.kill()
                self.process.wait()
                
        if self.master_fd is not None:
            os.close(self.master_fd)
            self.master_fd = None
```

**策略 3: 统一错误处理**

```python
# Before: 分散的错误处理
try:
    result = await some_operation()
except ValueError as e:
    print(f"Error: {e}")
except RuntimeError as e:
    print(f"Runtime error: {e}")
except Exception as e:
    print(f"Unknown error: {e}")

# After: 统一的错误处理器
class ErrorHandler:
    """统一错误处理器."""
    
    ERROR_MESSAGES = {
        ValueError: "输入错误: {message}",
        RuntimeError: "运行时错误: {message}",
        ConnectionError: "连接失败: {message}",
        TimeoutError: "操作超时，请重试",
    }
    
    @classmethod
    def handle(cls, error: Exception, context: str = "") -> str:
        """处理错误并返回用户友好的消息."""
        error_type = type(error)
        template = cls.ERROR_MESSAGES.get(
            error_type, 
            "发生错误: {message}"
        )
        return template.format(message=str(error))
    
    @classmethod
    async def handle_async(cls, error: Exception, stream: EventStream):
        """异步处理错误，发送事件."""
        message = cls.handle(error)
        stream.push(AgentEvent(
            type=AgentEventType.ERROR,
            error=message
        ))

# 使用

try:
    result = await some_operation()
except Exception as e:
    await ErrorHandler.handle_async(e, stream)
```
