### 2.1 重构 1: `frontend.py:render_event()` - 提取处理器类

**当前问题分析**:
- `render_event()` 使用 match-case 处理 12+ 种事件类型
- 每个 case 包含不同的处理逻辑（启动指示器、渲染内容、停止指示器等）
- 代码重复：`REASONING_TOKEN` 和 `CONTENT_TOKEN` 处理逻辑相似
- 复杂度 15，远超建议的 10

**重构方案 - 策略模式 + 处理器映射表**:

```python
# frontend.py - 重构后的架构

class EventHandler(ABC):
    """事件处理器抽象基类."""
    
    @abstractmethod
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        """处理事件."""
        pass


class MessageStartHandler(EventHandler):
    """处理 MESSAGE_START 事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        if frontend._indicator:
            frontend._indicator.stop()
        frontend._indicator = StreamingIndicator()
        frontend._indicator.start()
        frontend._reset_render_state()


class TokenHandler(EventHandler):
    """Token 处理器基类."""
    
    def __init__(self, is_reasoning: bool):
        self.is_reasoning = is_reasoning
        self.color = C.GRAY if is_reasoning else C.RESET
        self.prefix = "[Reasoning]: " if is_reasoning else "[Assistant]: "
        self.flag_attr = '_r_started' if is_reasoning else '_c_started'
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        if frontend._indicator:
            frontend._indicator.on_token()
        self._render(frontend, event.data)
    
    def _render(self, frontend: 'Frontend', content: str) -> None:
        flag = getattr(frontend, self.flag_attr)
        if not flag:
            # 如果需要，先换行
            if self.is_reasoning and frontend._c_started:
                print()
            print(f"{self.color}{self.prefix}{C.RESET}", end="", flush=True)
            setattr(frontend, self.flag_attr, True)
        print(f"{self.color}{content}{C.RESET}", end="", flush=True)


class MessageUpdateHandler(EventHandler):
    """处理 MESSAGE_UPDATE 事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        partial = getattr(event, 'partial_result', None)
        if not partial:
            return
        
        msg_type = partial.get('type')
        content = partial.get('content', '')
        
        if msg_type == 'reasoning':
            _render_reasoning(frontend, content)
        elif msg_type == 'content':
            _render_content(frontend, content)


def _render_reasoning(frontend: 'Frontend', content: str) -> None:
    """渲染推理内容."""
    if not frontend._r_started:
        print(f"{C.GRAY}[Reasoning]: {C.RESET}", end="", flush=True)
        frontend._r_started = True
    print(f"{C.GRAY}{content}{C.RESET}", end="", flush=True)


def _render_content(frontend: 'Frontend', content: str) -> None:
    """渲染内容."""
    if not frontend._c_started:
        if frontend._r_started:
            print()
        print(f"[Assistant]: ", end="", flush=True)
        frontend._c_started = True
    print(content, end="", flush=True)


class MessageEndHandler(EventHandler):
    """处理 MESSAGE_END 事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        if frontend._indicator:
            frontend._indicator.stop()
            frontend._indicator = None
        frontend._finalize_output()
        frontend._reset_render_state()


class ToolStartHandler(EventHandler):
    """处理 TOOL_EXECUTION_START 事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        frontend._tool_active = True
        tool_name = event.tool_name or "unknown"
        args = event.args
        args_str = _format_args(args)
        print(f"\n{C.YELLOW}[Tool: {tool_name}]{C.RESET}", end="", flush=True)
        if args_str:
            print(f" {C.DIM}{args_str}{C.RESET}", end="", flush=True)
        print(" → ", end="", flush=True)


def _format_args(args: Optional[Dict]) -> str:
    """格式化工具参数."""
    if not args:
        return ""
    args_str = str(args)
    if len(args_str) > 50:
        args_str = args_str[:50] + "..."
    return args_str


class ToolEndHandler(EventHandler):
    """处理 TOOL_EXECUTION_END 事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        frontend._tool_active = False
        is_error = event.is_error
        color = C.GRAY if is_error else C.GREEN
        result_str = _format_result(event.result)
        print(f"{color}{result_str}{C.RESET}")


def _format_result(result: Any) -> str:
    """格式化工具结果."""
    result_str = str(result) if result else "Done"
    if len(result_str) > 100:
        result_str = result_str[:100] + "..."
    return result_str


class StatsHandler(EventHandler):
    """处理 STATS 事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        stats = getattr(event, 'stats', None)
        if stats:
            frontend._stats = stats
            print(f"\n{C.CYAN}{frontend._format_stats(stats)}{C.RESET}")


class ErrorHandler(EventHandler):
    """处理 ERROR 事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        if frontend._indicator:
            frontend._indicator.stop()
            frontend._indicator = None
        error = getattr(event, 'error', 'Unknown error')
        print(f"\n{C.GRAY}[Error: {error}]{C.RESET}")


class SilentHandler(EventHandler):
    """静默处理生命周期事件."""
    
    def handle(self, frontend: 'Frontend', event: AgentEvent) -> None:
        pass  # 静默处理


# Frontend 类中的重构部分
class Frontend:
    """重构后的 Frontend - 使用处理器映射表."""
    
    def __init__(self, daemon: Daemon, session_manager=None):
        self.daemon = daemon
        self._session_manager = session_manager
        
        # 渲染状态
        self._r_started = False
        self._c_started = False
        self._tool_active = False
        self._prev_type: Optional[AgentEventType] = None
        self._stats: Dict[str, Any] = {}
        self._indicator: Optional[StreamingIndicator] = None
        
        # 初始化事件处理器映射表
        self._handlers = self._init_handlers()
    
    def _init_handlers(self) -> Dict[AgentEventType, EventHandler]:
        """初始化事件处理器映射表."""
        return {
            AgentEventType.MESSAGE_START: MessageStartHandler(),
            AgentEventType.REASONING_TOKEN: TokenHandler(is_reasoning=True),
            AgentEventType.CONTENT_TOKEN: TokenHandler(is_reasoning=False),
            AgentEventType.MESSAGE_UPDATE: MessageUpdateHandler(),
            AgentEventType.MESSAGE_END: MessageEndHandler(),
            AgentEventType.TOOL_EXECUTION_START: ToolStartHandler(),
            AgentEventType.TOOL_EXECUTION_END: ToolEndHandler(),
            AgentEventType.STATS: StatsHandler(),
            AgentEventType.ERROR: ErrorHandler(),
            AgentEventType.AGENT_START: SilentHandler(),
            AgentEventType.AGENT_END: SilentHandler(),
            AgentEventType.TURN_START: SilentHandler(),
            AgentEventType.TURN_END: SilentHandler(),
        }
    
    def render_event(self, event: AgentEvent) -> None:
        """渲染单个事件 - 简化后主入口 (目标 < 30 行)."""
        # 类型切换时添加分隔符
        if self._need_separator(event.type):
            print("\n")
        self._prev_type = event.type
        
        # 使用映射表分发处理
        handler = self._handlers.get(event.type)
        if handler:
            handler.handle(self, event)
        else:
            log(f"Unknown event type: {event.type}")
```

**预期效果**:
- `render_event()` 从 ~88 行降至 ~15 行
- 复杂度从 15 降至 ~5
- 每个处理器职责单一，易于测试
- 新增事件类型只需添加处理器，不修改主函数

```python
# daemon.py - 重构后的架构

class SessionManager:
    """会话管理器 - 独立管理对话历史."""
    
    def __init__(self, history_limit: int = 10):
        self._history_limit = history_limit
        self._messages: List[Dict[str, str]] = [
            {"role": "system", "content": "You are a helpful assistant."}
        ]
    
    @property
    def messages(self) -> List[Dict[str, str]]:
        """当前对话历史 (只读)."""
        return self._messages.copy()
    
    def add_user_message(self, content: str) -> None:
        """添加用户消息."""
        self._messages.append({"role": "user", "content": content})
    
    def add_assistant_message(self, content: str) -> None:
        """添加助手消息."""
        if content:
            self._messages.append({"role": "assistant", "content": content})
    
    def trim_history(self) -> None:
        """修剪历史消息，保留 system 消息和最近的 history_limit 条."""
        max_len = self._history_limit + 1  # +1 for system message
        if len(self._messages) > max_len:
            self._messages = [self._messages[0]] + self._messages[-self._history_limit:]
    
    def clear(self) -> None:
        """清除对话历史，只保留 system 消息."""
        self._messages = [self._messages[0]]


class ChatStreamProcessor:
    """聊天流处理器 - 处理流式响应和统计收集."""
    
    def __init__(self, session_manager: SessionManager):
        self._session = session_manager
        self._content_parts: List[str] = []
        self._reasoning_parts: List[str] = []
        self._stats = TokenStats()
    
    async def process(
        self, 
        stream: EventStream[AgentEvent, Any],
        output_stream: EventStream[AgentEvent, ChatResult]
    ) -> ChatResult:
        """处理流式响应并返回结果."""
        self._stats.start_time = time.time()
        
        async for event in stream:
            # 转发事件
            output_stream.push(event)
            
            # 收集内容
            if event.type == AgentEventType.MESSAGE_UPDATE:
                self._collect_message_update(event)
            
            # 收集最终结果
            if event.type == AgentEventType.TURN_END:
                self._collect_turn_end(event)
        
        self._stats.finalize()
        return self._build_result()
    
    def _collect_message_update(self, event: AgentEvent) -> None:
        """收集消息更新."""
        partial = getattr(event, 'partial_result', None)
        if not partial:
            return
        
        msg_type = partial.get('type')
        content = partial.get('content', '')
        
        if msg_type == 'reasoning':
            self._reasoning_parts.append(content)
            self._stats.on_token()
            self._stats.r_tokens += len(content) // 4
        elif msg_type == 'content':
            self._content_parts.append(content)
            self._stats.on_token()
            self._stats.c_tokens += len(content) // 4
    
    def _collect_turn_end(self, event: AgentEvent) -> None:
        """收集回合结束消息."""
        if hasattr(event, 'message') and event.message:
            msg = event.message
            if hasattr(msg, 'content'):
                for content in msg.content:
                    if hasattr(content, 'text'):
                        self._content_parts.append(content.text)
    
    def _build_result(self) -> ChatResult:
        """构建最终结果."""
        content = ''.join(self._content_parts)
        reasoning = ''.join(self._reasoning_parts)
        
        self._stats.tokens = self._stats.r_tokens + self._stats.c_tokens
        
        # 保存到历史
        self._session.add_assistant_message(content)
        
        return ChatResult(
            content=content,
            messages=[{"role": "assistant", "content": content}],
            reasoning=reasoning,
        )
    
    @property
    def stats(self) -> TokenStats:
        """获取统计信息."""
        return self._stats
    
    @property
    def partial_content(self) -> str:
        """获取当前已收集的部分内容."""
        return ''.join(self._content_parts)


class Daemon:
    """重构后的 Daemon - 精简版本."""
    
    def __init__(
        self,
        provider_api: str = "openai",
        model: Optional[str] = None,
        history_limit: int = 10,
        tool_registry: Optional["ToolRegistry"] = None,
    ):
        self._provider_api = provider_api
        self._model = model
        self._tool_registry = tool_registry
        
        # 初始化组件
        self._session = SessionManager(history_limit)
        self._provider = ProviderRegistry.get(provider_api)
    
    @property
    def model(self) -> str:
        """当前模型名称."""
        return self._model or "unknown"
    
    @property
    def messages(self) -> List[Dict[str, str]]:
        """当前对话历史 (只读)."""
        return self._session.messages
    
    def switch_model(self, name: str) -> None:
        """切换模型."""
        self._model = name
    
    def clear_history(self) -> None:
        """清除对话历史."""
        self._session.clear()
    
    def chat(
        self, text: str, signal: Optional["AbortSignal"] = None
    ) -> EventStream[AgentEvent, ChatResult]:
        """发送消息并获取流式响应 - 简化版 (目标 < 50 行)."""
        stream = EventStream[AgentEvent, ChatResult]()
        
        async def _stream():
            try:
                # 1. 验证前置条件
                provider = self._get_provider()
                if not provider:
                    self._emit_error(stream, f"Provider not found: {self._provider_api}")
                    return
                
                if signal and signal.aborted:
                    self._emit_error(stream, f"Aborted: {signal.reason}")
                    return
                
                # 2. 准备会话
                self._session.trim_history()
                self._session.add_user_message(text)
                
                # 3. 调用 Provider
                model = self._model or getattr(provider, 'default_model', 'gpt-3.5-turbo')
                tools = self._get_tools()
                
                stream.push(AgentEvent(type=AgentEventType.AGENT_START))
                stream.push(AgentEvent(type=AgentEventType.TURN_START))
                
                provider_stream = await provider.stream_chat(
                    model=model,
                    messages=self._session.messages,
                    tools=tools,
                    signal=signal,
                )
                
                # 4. 处理流式响应
                processor = ChatStreamProcessor(self._session)
                result = await processor.process(provider_stream, stream)
                
                # 5. 发送统计和结束事件
                self._emit_stats(stream, processor.stats)
                stream.push(AgentEvent(type=AgentEventType.AGENT_END))
                stream.end(result)
                
            except asyncio.CancelledError:
                self._handle_cancel(stream, processor)
                raise
            except Exception as e:
                self._emit_error(stream, str(e))
                stream.error(e)
        
        asyncio.create_task(_stream())
        return stream
    
    def _get_provider(self) -> Optional[Any]:
        """获取当前 Provider."""
        if self._provider is None:
            self._provider = ProviderRegistry.get(self._provider_api)
        return self._provider
    
    def _get_tools(self) -> Optional[List[Dict]]:
        """获取工具定义."""
        if self._tool_registry:
            return self._tool_registry.to_openai_tools()
        return None
    
    def _emit_error(self, stream: EventStream, message: str) -> None:
        """发送错误事件."""
        stream.push(AgentEvent(type=AgentEventType.ERROR, error=message))
        stream.end(ChatResult(content="", messages=[]))
    
    def _emit_stats(self, stream: EventStream, stats: TokenStats) -> None:
        """发送统计事件."""
        stream.push(AgentEvent(
            type=AgentEventType.STATS,
            stats={
                "tokens": stats.tokens,
                "r_tokens": stats.r_tokens,
                "c_tokens": stats.c_tokens,
                "elapsed": stats.elapsed,
                "tps": stats.tps,
                "ttft": stats.ttft,
            }
        ))
    
    def _handle_cancel(
        self, 
        stream: EventStream, 
        processor: Optional[ChatStreamProcessor]
    ) -> None:
        """处理取消."""
        content = processor.partial_content if processor else ""
        if content:
            self._session.add_assistant_message(content)
        stream.push(AgentEvent(type=AgentEventType.ERROR, error="Cancelled by user"))
        stream.end(ChatResult(content=content, messages=[], finish_reason="cancelled"))
```

**预期效果**:
- `chat()` 从 ~170 行降至 ~50 行
- 复杂度从 20 降至 ~8
- `SessionManager` 独立管理历史，可复用
- `ChatStreamProcessor` 处理流逻辑，可测试
- 错误处理统一通过 `_emit_error` 等方法

```python
# error_handler.py - 增强版

class ErrorHandler:
    """统一错误处理器 - 支持上下文和 Provider 特定错误."""
    
    # 错误模式映射 (保持不变)
    ERROR_PATTERNS: Dict[ErrorCategory, list[str]] = {...}
    FRIENDLY_MESSAGES: Dict[ErrorCategory, tuple[str, str]] = {...}
    
    @classmethod
    def handle(
        cls, 
        error: Exception, 
        context: str = "",
        provider: str = "",
        fallback: str = "发生未知错误"
    ) -> FriendlyError:
        """处理错误并返回友好信息."""
        category = cls.categorize(error)
        title, suggestion = cls.FRIENDLY_MESSAGES[category]
        
        # 根据 Provider 定制建议
        if provider and category == ErrorCategory.AUTH:
            suggestion = f"请检查 {provider.upper()}_API_KEY 环境变量。"
        
        return FriendlyError(
            category=category,
            original=str(error),
            title=title,
            message=str(error),
            suggestion=suggestion,
        )
    
    @classmethod
    def create_event(
        cls, 
        error: Exception, 
        context: str = "",
        provider: str = ""
    ) -> AgentEvent:
        """创建错误事件."""
        friendly = cls.handle(error, context, provider)
        return AgentEvent(
            type=AgentEventType.ERROR,
            error=f"[{context}] {friendly.title}: {friendly.message}" if context else f"{friendly.title}: {friendly.message}",
            suggestion=friendly.suggestion,
        )
    
    @classmethod
    def format_for_display(cls, error: Exception, context: str = "") -> str:
        """格式化为显示字符串."""
        friendly = cls.handle(error, context)
        lines = [
            f"\n⚠️  {friendly.title}",
            f"   {friendly.message[:100]}",
            "",
            f"💡 {friendly.suggestion}",
        ]
        return "\n".join(lines)
```

```python
        return None


class ResponseBuilder:
    """响应构建器 - 从流事件构建 AssistantMessage."""
    
    def __init__(self):
        self.content_parts: List[Any] = []
        self.current_text = ""
        self.finish_reason: Optional[str] = None
        self.message_started = False
    
    def process_event(self, event: AgentEvent) -> None:
        """处理单个事件."""
        if event.type == AgentEventType.MESSAGE_START:
            self.message_started = True
        
        elif event.type == AgentEventType.MESSAGE_UPDATE:
            self._process_update(event)
        
        elif event.type == AgentEventType.MESSAGE_END:
            self._process_end(event)
        
        elif event.type == AgentEventType.TURN_END:
            self._process_turn_end(event)
    
    def _process_update(self, event: AgentEvent) -> None:
        """处理消息更新."""
        partial = event.partial_result or {}
        if partial.get("type") == "content":
            self.current_text += partial.get("content", "")
    
    def _process_end(self, event: AgentEvent) -> None:
        """处理消息结束."""
        if event.message and isinstance(event.message, AssistantMessage):
            self.content_parts = list(event.message.content)
            self.finish_reason = event.message.stop_reason
    
    def _process_turn_end(self, event: AgentEvent) -> None:
        """处理回合结束."""
        if event.message and isinstance(event.message, AssistantMessage):
            self.content_parts = list(event.message.content)
            self.finish_reason = event.message.stop_reason
    
    def try_extract_from_result(self, result: Any) -> None:
        """尝试从流结果提取消息."""
        if result and result.messages:
            for msg in result.messages:
                if isinstance(msg, AssistantMessage):
                    for content in msg.content:
                        if isinstance(content, ToolCallContent):
                            self.content_parts.append(content)
                    if msg.stop_reason:
                        self.finish_reason = msg.stop_reason
    
    def build(self) -> AssistantMessage:
        """构建最终的 AssistantMessage."""
        if self.current_text and not self.content_parts:
            self.content_parts.append(TextContent(text=self.current_text))
        
        return AssistantMessage(
            content=self.content_parts,
            stop_reason=self.finish_reason or "stop",
        )


async def _stream_assistant_response(
    context: List[AgentMessage],
    config: AgentLoopConfig,
    signal: Optional[AbortSignal],
    emit: AgentEventSink,
) -> AssistantMessage:
    """流式获取助手响应 - 简化版 (目标 < 50 行)."""
    # 1. 准备 Provider
    provider = ProviderAdapter.get_provider(config)
    messages = ProviderAdapter.convert_messages(context, config.system_prompt)
    tools = ProviderAdapter.prepare_tools(config.tool_registry)
    
    # 2. 调用 Provider
    stream = await provider.stream_chat(
        model=config.model,
        messages=messages,
        tools=tools,
        signal=signal,
        temperature=config.temperature,
        max_tokens=config.max_tokens,
    )
    
    # 3. 处理流并构建响应
    builder = ResponseBuilder()
    async for event in stream:
        await emit(event)
        builder.process_event(event)
    
    # 4. 尝试从结果提取额外信息
    try:
        result = await stream.result()
        builder.try_extract_from_result(result)
    except Exception:
        pass
    
    return builder.build()
```

**预期效果**:
- `_stream_assistant_response()` 从 ~100 行降至 ~25 行
- 复杂度从 12 降至 ~5
- `ProviderAdapter` 可复用
- `ResponseBuilder` 可独立测试
