## 执行摘要

基于对 pi-mono Core Runtime (TypeScript) 的深度分析，本方案为 tui-chatbot (Python) 设计了一套渐进式架构升级路径。方案借鉴 pi-mono 的核心设计模式，但适配 Python 生态，重点关注可扩展性、类型安全和流式处理。

### 本轮迭代目标 (N=1)

构建**三大核心基础设施**:
1. **分层事件系统** - 从5个事件扩展到完整生命周期覆盖
2. **Provider Registry 架构** - 解耦 LLM 提供商依赖
3. **工具调用框架** - 支持函数调用能力

---

## 1. 核心设计原则

### 1.1 借鉴自 pi-mono 的模式

| pi-mono 模式 | Python 适配方案 |
|-------------|----------------|
| EventStream (流式+Promise) | `EventStream[T, R]` 类，支持 `async for` 和 `await result()` |
| Provider Registry | `ProviderRegistry` 类，运行时动态注册 |
| Lazy Provider Loading | `importlib` 延迟导入 + 模块级缓存 |
| Agent Loop (双层循环) | `AgentLoop` 类，外层处理 follow-up，内层处理工具 |
| Tool Execution (三阶段) | `prepare → execute (并行) → finalize` 管道 |

### 1.2 Python 生态适配

- **类型系统**: 使用 `pydantic` 替代 TypeBox，同时获得运行时验证
- **异步模式**: `asyncio` + `async/await`，与 pi-mono 类似
- **取消信号**: `AbortController` 模式，类似 JavaScript 的 AbortSignal
- **泛型**: Python 3.11+ `TypeVar` 和 `Generic`

### 4.1 EventStream 类 (核心基础设施)

**文件**: `src/tui_chatbot/core/event_stream.py`

```python
import asyncio
from typing import AsyncIterator, Generic, Optional, TypeVar

T = TypeVar("T")
R = TypeVar("R")

class EventStream(Generic[T, R]):
    """
    双接口事件流：支持 async for 迭代和 await result() 获取结果
    
    用法：
        stream = daemon.chat("hello")
        async for event in stream:
            print(event)
        result = await stream.result()
    """
    
    def __init__(self):
        self._queue: asyncio.Queue[T] = asyncio.Queue()
        self._done = asyncio.Event()
        self._result_future: asyncio.Future[R] = asyncio.Future()
        self._error: Optional[Exception] = None
    
    def push(self, event: T) -> None:
        """推送事件到流中"""
        if self._done.is_set():
            return
        self._queue.put_nowait(event)
    
    def end(self, result: Optional[R] = None) -> None:
        """结束流并设置结果"""
        if not self._done.is_set():
            if result is not None and not self._result_future.done():
                self._result_future.set_result(result)
            elif not self._result_future.done():
                self._result_future.set_result(None)  # type: ignore
            self._done.set()
    
    def error(self, exc: Exception) -> None:
        """以错误结束流"""
        if not self._done.is_set():
            self._error = exc
            if not self._result_future.done():
                self._result_future.set_exception(exc)
            self._done.set()
    
    async def result(self) -> R:
        """获取最终结果 (Promise 风格)"""
        await self._done.wait()
        if self._error:
            raise self._error
        return await self._result_future
    
    def __aiter__(self) -> AsyncIterator[T]:
        return self
    
    async def __anext__(self) -> T:
        """获取下一个事件 (迭代风格)"""
        while True:
            if self._done.is_set() and self._queue.empty():
                raise StopAsyncIteration
            try:
                return self._queue.get_nowait()
            except asyncio.QueueEmpty:
                try:
                    await asyncio.wait_for(self._done.wait(), timeout=0.1)
                    if self._queue.empty():
                        raise StopAsyncIteration
                except asyncio.TimeoutError:
                    continue
```

from abc import ABC, abstractmethod
from typing import Dict, List, Optional, Callable, Awaitable
from dataclasses import dataclass
from ..core.event_stream import EventStream
from ..models.events import AgentEvent, ChatResult

class Provider(ABC):
    """LLM 提供商抽象基类"""
    
    @property
    @abstractmethod
    def name(self) -> str:
        """提供商名称"""
        pass
    
    @property
    @abstractmethod
    def api_type(self) -> str:
        """API 类型标识"""
        pass
    
    @abstractmethod
    async def stream_chat(
        self, 
        model: str,
        messages: List[dict],
        tools: Optional[List[dict]] = None,
        signal: Optional[AbortSignal] = None
    ) -> EventStream[AgentEvent, ChatResult]:
        """
        流式对话接口
        
        返回 EventStream，推送 AgentEvent 事件，
        最终返回 ChatResult
        """
        pass
    
    @abstractmethod
    async def list_models(self) -> List[str]:
        """列出可用模型"""
        pass


class LazyProvider(Provider):
    """延迟加载的提供商包装器"""
    
    def __init__(self, loader: Callable[[], Awaitable[Provider]]):
        self._loader = loader
        self._provider: Optional[Provider] = None
    
    async def _ensure_loaded(self) -> Provider:
        if self._provider is None:
            self._provider = await self._loader()
        return self._provider
    
    @property
    def name(self) -> str:
        return "lazy"  # 加载前未知
    
    @property
    def api_type(self) -> str:
        return "lazy"
    
    async def stream_chat(self, model, messages, tools=None, signal=None):
        provider = await self._ensure_loaded()
        return await provider.stream_chat(model, messages, tools, signal)
    
    async def list_models(self):
        provider = await self._ensure_loaded()
        return await provider.list_models()


class ProviderRegistry:
    """LLM 提供商注册表 - 运行时动态注册"""
    
    _providers: Dict[str, Provider] = {}
    
    @classmethod
    def register(cls, api_type: str, provider: Provider) -> None:
        """注册提供商"""
        cls._providers[api_type] = provider
    
    @classmethod
    def get(cls, api_type: str) -> Optional[Provider]:
        """获取提供商"""
        return cls._providers.get(api_type)
    
    @classmethod
    def list(cls) -> List[str]:
        """列出所有已注册 API 类型"""
        return list(cls._providers.keys())
    
    @classmethod
    def clear(cls) -> None:
        """清空注册表 (主要用于测试)"""
        cls._providers.clear()

### 4.3 Agent Loop (双层循环)

**文件**: `src/tui_chatbot/agent/loop.py`

```python
from typing import List, Optional, Callable, Awaitable
from dataclasses import dataclass
from ..core.event_stream import EventStream
from ..core.abort_controller import AbortSignal
from ..models.events import AgentEvent, AgentEventType
from ..models.messages import AgentMessage, AssistantMessage, ToolCallMessage, ToolResultMessage
from .tool import ToolRegistry, ToolExecutionMode

AgentEventSink = Callable[[AgentEvent], Awaitable[None]]

@dataclass
class AgentLoopConfig:
    """Agent 循环配置"""
    model: str
    system_prompt: str
    tool_registry: ToolRegistry
    tool_execution_mode: ToolExecutionMode = ToolExecutionMode.PARALLEL
    max_iterations: int = 10  # 防止无限循环

async def agent_loop(
    prompt: AgentMessage,
    history: List[AgentMessage],
    config: AgentLoopConfig,
    signal: Optional[AbortSignal] = None,
    emit: Optional[AgentEventSink] = None
) -> EventStream[AgentEvent, List[AgentMessage]]:
    """
    启动 Agent 循环
    
    双层循环结构：
    - 外层：处理 follow-up 消息
    - 内层：处理单轮对话 (助手响应 + 工具调用)
    """
    stream = EventStream[AgentEvent, List[AgentMessage]]()
    
    async def _run():
        try:
            all_new_messages: List[AgentMessage] = [prompt]
            context = history + [prompt]
            
            await _emit(AgentEvent(type=AgentEventType.AGENT_START))
            
            iteration = 0
            while iteration < config.max_iterations:
                iteration += 1
                
                # 内层循环：处理工具调用
                has_more_tool_calls = True
                while has_more_tool_calls:
                    await _emit(AgentEvent(type=AgentEventType.TURN_START))
                    
                    # 流式助手响应
                    assistant_msg = await _stream_assistant_response(
                        context, config, signal, _emit
                    )
                    all_new_messages.append(assistant_msg)
                    
                    # 检查工具调用
                    tool_calls = _extract_tool_calls(assistant_msg)
                    has_more_tool_calls = len(tool_calls) > 0
                    
                    if has_more_tool_calls:
                        # 执行工具调用
                        tool_results = await _execute_tool_calls(
                            tool_calls, config.tool_registry, signal, _emit
                        )
                        context.extend(tool_results)
                        all_new_messages.extend(tool_results)
                    
                    await _emit(AgentEvent(
                        type=AgentEventType.TURN_END,
                        message=assistant_msg,
                        tool_results=tool_results if has_more_tool_calls else []
                    ))
                
                # 检查是否需要继续 (follow-up)
                # 简化版：暂时不支持 follow-up
                break
            
            await _emit(AgentEvent(
                type=AgentEventType.AGENT_END,
                messages=all_new_messages
            ))
            stream.end(all_new_messages)
            
        except Exception as e:
            await _emit(AgentEvent(type=AgentEventType.ERROR, error=str(e)))
            stream.error(e)
    
    async def _emit(event: AgentEvent):
        stream.push(event)
        if emit:
            await emit(event)
    
    # 启动后台任务
    asyncio.create_task(_run())
    return stream

async def _execute_tool_calls(
    tool_calls: List[ToolCallMessage],
    registry: ToolRegistry,
    signal: Optional[AbortSignal],
    emit: AgentEventSink
) -> List[ToolResultMessage]:
    """
    执行工具调用 - 三阶段模式
    Phase 1: Sequential preflight (验证)
    Phase 2: Concurrent execution
    Phase 3: Sequential finalization
    """
    results: List[ToolResultMessage] = []
    
    # Phase 1: Preflight (串行验证)
    prepared = []
    for tc in tool_calls:
        await emit(AgentEvent(
            type=AgentEventType.TOOL_EXECUTION_START,
            tool_call_id=tc.id,
            tool_name=tc.name,
            args=tc.arguments
        ))
        
        tool = registry.get(tc.name)
        if not tool:
            results.append(ToolResultMessage(
                role="tool_result",
                tool_call_id=tc.id,
                content=f"Tool {tc.name} not found",
                is_error=True
            ))
            continue
        
        # 验证参数
        try:
            validated = tool.validate_params(tc.arguments)
            prepared.append((tc, tool, validated))
        except Exception as e:
            results.append(ToolResultMessage(
                role="tool_result",
                tool_call_id=tc.id,
                content=f"Parameter validation failed: {e}",
                is_error=True
            ))
    
    # Phase 2: Concurrent execution
    async def execute_one(tc, tool, validated):
        try:
            result = await tool.execute(validated, signal)
            return (tc, result, False)
        except Exception as e:
            return (tc, str(e), True)
    
    executed = await asyncio.gather(*[
        execute_one(tc, tool, validated) 
        for tc, tool, validated in prepared
    ])
    
    # Phase 3: Finalization (按原始顺序)
    for tc, result, is_error in executed:
        tool_result_msg = ToolResultMessage(
            role="tool_result",
            tool_call_id=tc.id,
            tool_name=tc.name,
            content=result if not is_error else f"Error: {result}",
            is_error=is_error
        )
        results.append(tool_result_msg)
        
        await emit(AgentEvent(
            type=AgentEventType.TOOL_EXECUTION_END,
            tool_call_id=tc.id,
            tool_name=tc.name,
            result=result,
            is_error=is_error
        ))
    
    return results
```

from abc import ABC, abstractmethod
from typing import Dict, List, Optional, Callable, Awaitable, Any
from enum import Enum, auto
from pydantic import BaseModel, Field
from dataclasses import dataclass

class ToolExecutionMode(Enum):
    SEQUENTIAL = auto()
    PARALLEL = auto()

class ToolParameters(BaseModel):
    """工具参数基类 - 子类定义具体参数"""
    pass

class ToolResult(BaseModel):
    """工具执行结果"""
    content: str
    is_error: bool = False
    details: Dict[str, Any] = Field(default_factory=dict)

@dataclass
class ToolCall:
    """工具调用请求"""
    id: str
    name: str
    arguments: Dict[str, Any]

class Tool(ABC):
    """抽象工具基类"""
    
    @property
    @abstractmethod
    def name(self) -> str:
        """工具名称"""
        pass
    
    @property
    @abstractmethod
    def description(self) -> str:
        """工具描述 (用于 LLM)"""
        pass
    
    @property
    @abstractmethod
    def parameters(self) -> type[ToolParameters]:
        """参数模型类"""
        pass
    
    @abstractmethod
    async def execute(
        self, 
        params: ToolParameters,
        signal: Optional[AbortSignal] = None
    ) -> ToolResult:
        """执行工具"""
        pass
    
    def validate_params(self, args: Dict[str, Any]) -> ToolParameters:
        """验证参数 (使用 Pydantic)"""
        return self.parameters(**args)
    
    def to_openai_schema(self) -> Dict[str, Any]:
        """转换为 OpenAI 工具格式"""
        schema = self.parameters.model_json_schema()
        return {
            "type": "function",
            "function": {
                "name": self.name,
                "description": self.description,
                "parameters": schema
            }
        }


class ToolRegistry:
    """工具注册表"""
    
    def __init__(self):
        self._tools: Dict[str, Tool] = {}
    
    def register(self, tool: Tool) -> None:
        """注册工具"""
        self._tools[tool.name] = tool
    
    def get(self, name: str) -> Optional[Tool]:
        """获取工具"""
        return self._tools.get(name)
    
    def list(self) -> List[str]:
        """列出所有工具名称"""
        return list(self._tools.keys())
    
    def to_openai_tools(self) -> List[Dict[str, Any]]:
        """转换为 OpenAI 工具列表"""
        return [tool.to_openai_schema() for tool in self._tools.values()]


# ═══════════════════════════════════════════════════════════════

class GetCurrentTimeParams(ToolParameters):
    """获取当前时间参数"""
    timezone: Optional[str] = Field(
        default="UTC",
        description="时区，如 'UTC', 'Asia/Shanghai'"
    )

class GetCurrentTimeTool(Tool):
    """获取当前时间工具"""
    
    @property
    def name(self) -> str:
        return "get_current_time"
    
    @property
    def description(self) -> str:
        return "获取当前日期和时间"
    
    @property
    def parameters(self) -> type[ToolParameters]:
        return GetCurrentTimeParams
    
    async def execute(self, params: GetCurrentTimeParams, signal=None) -> ToolResult:
        from datetime import datetime
        import pytz
        
        try:
            tz = pytz.timezone(params.timezone)
            now = datetime.now(tz)
            return ToolResult(
                content=now.strftime("%Y-%m-%d %H:%M:%S %Z"),
                details={"timezone": params.timezone}
            )
        except Exception as e:
            return ToolResult(
                content=f"Error: {e}",
                is_error=True
            )


def create_default_tool_registry() -> ToolRegistry:
    """创建包含默认工具的注册表"""
    registry = ToolRegistry()
    registry.register(GetCurrentTimeTool())
    return registry
