```python
class Frontend:
    """Frontend - 处理 UI 渲染 (增强版)."""

    def __init__(self, daemon: Daemon):
        self.daemon = daemon

        # 渲染状态
        self._r_started = False
        self._c_started = False
        self._tool_active = False
        self._prev_type: Optional[AgentEventType] = None
        self._stats: Dict[str, Any] = {}

        # 新增：流式指示器
        self._indicator: Optional[StreamingIndicator] = None

    def render_event(self, event: AgentEvent) -> None:
        """渲染单个事件 (增强版)."""

        # 处理不同类型事件
        match event.type:
            case AgentEventType.MESSAGE_START:
                # 启动加载指示器
                self._indicator = StreamingIndicator()
                self._indicator.start()
                self._reset_render_state()

            case AgentEventType.CONTENT_TOKEN:
                # 更新指示器并显示 token
                if self._indicator:
                    self._indicator.on_token()
                self.on_token(event.data, is_reasoning=False)

            case AgentEventType.REASONING_TOKEN:
                # 更新指示器但不计数（推理 token 单独统计）
                if self._indicator:
                    self._indicator.on_token()  # 或者单独统计推理 token
                self.on_token(event.data, is_reasoning=True)

            case AgentEventType.MESSAGE_END:
                # 停止指示器并显示最终统计
                if self._indicator:
                    self._indicator.stop()
                    self._indicator = None
                self._finalize_output()

            case AgentEventType.ERROR:
                # 错误时停止指示器
                if self._indicator:
                    self._indicator.stop()
                    self._indicator = None
                self._handle_error(event)

            # ... 其他事件处理保持不变
```

```python
class SessionManager:
    """会话管理器 (增强版)."""

    # ... 现有方法保持不变 ...

    def save_current(self) -> bool:
        """保存当前会话.

        Returns:
            bool: 是否成功保存
        """
        if self._current_session is None:
            return False

        self._storage.save(self._current_session)
        return True

    def rename(self, session_id: str, new_title: str) -> bool:
        """重命名会话.

        Args:
            session_id: 会话 ID
            new_title: 新标题

        Returns:
            bool: 是否成功
        """
        session = self._storage.load(session_id)
        if not session:
            return False

        session.metadata.title = new_title
        session.metadata.updated_at = datetime.now()
        self._storage.save(session)

        # 如果重命名的是当前会话，同步更新
        if (self._current_session and
            self._current_session.metadata.id == session_id):
            self._current_session = session

        return True

    def get_session_info(self, session_id: str) -> Optional[Dict[str, Any]]:
        """获取会话信息摘要.

        Returns:
            Dict 包含: id, title, message_count, model, updated_at
        """
        session = self._storage.load(session_id)
        if not session:
            return None

        return {
            "id": session.metadata.id,
            "title": session.metadata.title,
            "message_count": len(session.messages),
            "model": session.metadata.model,
            "created_at": session.metadata.created_at,
            "updated_at": session.metadata.updated_at,
        }
```

```python
            self._show_config_item(key)
        else:
            # 修改配置
            self._set_config(key, args[1])

    def _show_all_config(self) -> None:
        """显示所有配置."""
        config = self._config_manager.get()

        print(f"\n{C.BOLD}Current Configuration:{C.RESET}\n")

        for key, (desc, _) in self.CONFIG_ITEMS.items():
            value = getattr(config, key)
            print(f"  {C.CYAN}{key:<20}{C.RESET} {value}")
            print(f"  {'':20} {C.DIM}{desc}{C.RESET}")
            print()

        config_file = self._config_manager._config_file
        print(f"{C.DIM}Config file: {config_file}{C.RESET}")

    def _show_config_item(self, key: str) -> None:
        """显示单项配置."""
        config = self._config_manager.get()
        value = getattr(config, key)
        desc, type_ = self.CONFIG_ITEMS[key]

        print(f"\n{C.BOLD}{key}{C.RESET}")
        print(f"  Value: {C.CYAN}{value}{C.RESET}")
        print(f"  Type:  {type_.__name__}")
        print(f"  Desc:  {desc}")

    def _set_config(self, key: str, value_str: str) -> None:
        """设置配置值."""
        _, type_ = self.CONFIG_ITEMS[key]

        try:
            # 类型转换
            if type_ == bool:
                value = value_str.lower() in ("true", "1", "yes", "on")
            elif type_ == int:
                value = int(value_str)
            elif type_ == float:
                value = float(value_str)
            else:
                value = value_str

            self._config_manager.update(**{key: value})
            print(f"✓ {key} = {value}")

        except ValueError as e:
            print(f"✗ Invalid value: {e}")
            print(f"  Expected type: {type_.__name__}")
```

#### 2.4.1 缓存索引设计

```python
class CachedSearchEngine(MessageSearchEngine):
    """带缓存的消息搜索引擎."""

    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self._session_cache: Dict[str, tuple[ChatSession, List[AgentMessage], datetime]] = {}
        self._cache_hits = 0
        self._cache_misses = 0

    def index_session(self, session: ChatSession, force: bool = False) -> None:
        """索引会话 (支持缓存).

        Args:
            session: 要索引的会话
            force: 强制重新索引，忽略缓存
        """
        session_id = session.metadata.id
        current_version = session.metadata.updated_at

        # 检查缓存
        if not force and session_id in self._session_cache:
            cached_session, cached_messages, cached_version = self._session_cache[session_id]
            if cached_version == current_version:
                # 缓存有效，直接使用
                self._cache_hits += 1
                self._index.extend([(session, msg) for msg in cached_messages])
                return

        # 缓存未命中，执行索引
        self._cache_misses += 1
        messages = list(session.messages)

        # 存储到缓存
        self._session_cache[session_id] = (session, messages, current_version)

        # 添加到索引
        for msg in messages:
            self._index.append((session, msg))

    def get_cache_stats(self) -> Dict[str, Any]:
        """获取缓存统计."""
        total = self._cache_hits + self._cache_misses
        hit_rate = self._cache_hits / total if total > 0 else 0

        return {
            "hits": self._cache_hits,
            "misses": self._cache_misses,
            "hit_rate": hit_rate,
            "cached_sessions": len(self._session_cache),
        }

    def clear_cache(self) -> None:
        """清空缓存."""
        self._session_cache.clear()
        self._cache_hits = 0
        self._cache_misses = 0
```

```python
@dataclass
class PaginationParams:
    """分页参数."""
    page_size: int = 20
    page_num: int = 1

    @property
    def offset(self) -> int:
        return (self.page_num - 1) * self.page_size

    @property
    def limit(self) -> int:
        return self.page_size


class PaginatedMessageLoader:
    """分页消息加载器."""

    def __init__(self, session: ChatSession, page_size: int = 20):
        self._session = session
        self._page_size = page_size
        self._total_messages = len(session.messages)

    def get_page(self, page_num: int) -> List[AgentMessage]:
        """获取指定页的消息."""
        params = PaginationParams(page_size=self._page_size, page_num=page_num)
        start = params.offset
        end = min(start + params.limit, self._total_messages)

        if start >= self._total_messages:
            return []

        return self._session.messages[start:end]

    def get_page_info(self, page_num: int) -> Dict[str, Any]:
        """获取分页信息."""
        total_pages = (self._total_messages + self._page_size - 1) // self._page_size

        return {
            "page_num": page_num,
            "page_size": self._page_size,
            "total_pages": total_pages,
            "total_messages": self._total_messages,
            "has_next": page_num < total_pages,
            "has_prev": page_num > 1,
        }
```

```python
class HelpCommand:
    """/help - 显示帮助（增强版，包含提示信息）."""

    META = {
        "pass_mode": "shlex",
        "name": "help",
        "help": "显示帮助信息",
        "args": ["[command]"],
    }

    async def run(self, argv: List[str]) -> None:
        """运行帮助命令."""
        if len(argv) > 1:
            # 显示特定命令的帮助
            cmd_name = argv[1].lstrip("/")
            self._show_command_help(cmd_name)
        else:
            # 显示完整帮助
            self._show_full_help()

    def _show_full_help(self) -> None:
        """显示完整帮助."""
        help_text = f"""
{C.BOLD}TUI Chatbot - Available Commands:{C.RESET}

Chat:
  <text>                    Send message to AI (default)

Session Management:
  /save                     Save current session
  /sessions [--json]        List all sessions
  /load <id>                Load a session by ID
  /history [page]           View message history (paginated)
  /clear                    Clear current session history

Search & Export:
  /search <keyword>         Search in session history
                            Options: --regex, --user-only, --assistant-only, --all
  /export [--json] [--all]  Export session(s) to file

Configuration:
  /model [name]             List or switch models
  /config [key] [value]     View/modify configuration

Other:
  /help [command]           Show this help or specific command help
  /quit, /exit, /q          Exit the program

{C.DIM}Tips:{C.RESET}
  • Press {C.CYAN}Ctrl+C{C.RESET} to interrupt generation
  • Use {C.CYAN}Tab{C.RESET} for command completion (if supported)
  • Type {C.CYAN}/{C.RESET} to see available commands
"""
        print(help_text)

    def _show_command_help(self, cmd_name: str) -> None:
        """显示特定命令的帮助."""
        if cmd_name not in COMMAND_COMPLETIONS:
            print(f"Unknown command: {cmd_name}")
            print(f"Type /help to see all commands")
            return

        meta = COMMAND_COMPLETIONS[cmd_name]

        print(f"\n{C.BOLD}/{cmd_name}{C.RESET}")
        print(f"  {meta['description']}")

        if "aliases" in meta:
            print(f"\n  Aliases: {', '.join(f'/{a}' for a in meta['aliases'])}")

        if meta.get("args"):
            print(f"\n  Usage: /{cmd_name} {' '.join(meta['args'])}")

        if meta.get("options"):
            print(f"\n  Options:")
            for opt in meta["options"]:
                print(f"    {opt}")

        if meta.get("examples"):
            print(f"\n  Examples:")
            for ex in meta["examples"]:
                print(f"    {ex}")

        print()
```

```python
class TestCachedSearchEngine:
    """测试缓存搜索引擎."""

    def test_cache_hit_on_same_session(self):
        """相同会话应使用缓存."""
        engine = CachedSearchEngine()
        session = self._create_test_session()

        # 第一次索引
        engine.index_session(session)
        assert engine._cache_misses == 1

        # 清空索引但保留缓存
        engine._index.clear()

        # 第二次索引（应命中缓存）
        engine.index_session(session)
        assert engine._cache_hits == 1

    def test_cache_invalidate_on_update(self):
        """会话更新后缓存应失效."""
        engine = CachedSearchEngine()
        session = self._create_test_session()

        engine.index_session(session)
        original_misses = engine._cache_misses

        # 模拟会话更新
        session.metadata.updated_at = datetime.now()

        # 重新索引应未命中缓存
        engine._index.clear()
        engine.index_session(session)
        assert engine._cache_misses == original_misses + 1
```
