## 2. 详细设计规范

### 2.1 问题修复设计

#### 2.1.1 修复 `/quit` 不退出问题

**问题分析**:
- `QuitCommand.run()` 抛出 `QuitException`
- `Shell.run()` 捕获 `QuitException` 后只是 `break`
- 但 `sys.exit(0)` 在 `finally` 块中，应该能退出
- 实际情况可能是事件循环或异步任务阻止退出

**修复方案**:

```python
# main.py - QuitCommand
class QuitCommand:
    async def run(self, argv: List[str]) -> None:
        print("Bye!")
        # 确保立即触发退出，不依赖异常传播
        sys.exit(0)

# 或者使用更明确的方式
class QuitCommand:
    META = {"pass_mode": "shlex", "immediate_exit": True}
    
    async def run(self, argv: List[str]) -> None:
        print("Bye!")
        raise SystemExit(0)
```

#### 2.1.2 修复 `/save` 失败问题

**问题分析**:
- `SaveCommand` 检查 `self._session_manager.current()`
- 集成测试显示 "No active session to save"
- 会话在初始化时创建，但可能未被正确设置

**修复方案**:

```python
# main.py - Shell.__init__
def __init__(self, cfg: Config):
    # ... 现有初始化 ...
    
    # 创建 SessionManager 并立即创建当前会话
    storage = SessionStorage()
    self.session_manager = SessionManager(storage)
    
    # 确保创建并激活初始会话
    initial_session = self.session_manager.create("Default", cfg.model)
    self.session_manager._current_session_id = initial_session.metadata.id
    
    # Frontend 需要引用 session_manager 进行同步
    self.frontend = Frontend(self.daemon, self.session_manager)
```

#### 2.1.3 修复 Assistant 内容缺失问题

**问题分析**:
- 集成测试显示第二条消息只有推理过程，没有 `[Assistant]` 内容
- 可能是 `_c_started` 状态未正确重置
- 或 `MESSAGE_UPDATE` 事件中的内容类型判断错误

**修复方案**:

```python
# frontend.py - _handle_message_update
# 添加更健壮的内容类型检测

def _handle_message_update(self, event: AgentEvent) -> None:
    """处理消息更新事件 - 增强版."""
    partial = getattr(event, "partial_result", None)
    if not partial:
        return

    msg_type = partial.get("type")
    content = partial.get("content", "")
    
    # 添加调试日志
    log(f"MESSAGE_UPDATE: type={msg_type}, content_len={len(content)}, "
        f"r_started={self._r_started}, c_started={self._c_started}")

    if msg_type == "reasoning":
        if not self._r_started:
            print(f"{C.GRAY}[Reasoning]: {C.RESET}", end="", flush=True)
            self._r_started = True
        print(f"{C.GRAY}{content}{C.RESET}", end="", flush=True)

    elif msg_type == "content":
        # 强制重置检查 - 如果从未开始内容且已有推理，强制开始
        if not self._c_started:
            # 检查是否需要换行（前面有推理输出）
            if self._r_started:
                print()  # 换行分隔
            print(f"[Assistant]: ", end="", flush=True)
            self._c_started = True
        print(content, end="", flush=True)
        
    # 其他类型记录但不处理
    else:
        log(f"Unknown message type: {msg_type}")
```

### 2.2 代码重构设计

#### 2.2.1 重构 `render_event()` - 处理器映射模式

**当前问题**:
- `render_event()` 使用 match-case，约 85 行
- 处理 12 种不同事件类型，逻辑集中
- 难以测试和维护

**重构目标**:

```python
# frontend.py - 重构后的 render_event
class Frontend:
    def __init__(self, daemon: Daemon, session_manager=None):
        # ... 现有初始化 ...
        
        # 初始化事件处理器映射表
        self._event_handlers = self._init_event_handlers()
    
    def _init_event_handlers(self) -> Dict[AgentEventType, Callable]:
        """初始化事件处理器映射表."""
        return {
            AgentEventType.MESSAGE_START: self._handle_message_start,
            AgentEventType.REASONING_TOKEN: self._handle_reasoning_token,
            AgentEventType.CONTENT_TOKEN: self._handle_content_token,
            AgentEventType.MESSAGE_UPDATE: self._handle_message_update,
            AgentEventType.MESSAGE_END: self._handle_message_end,
            AgentEventType.TOOL_EXECUTION_START: self._handle_tool_start,
            AgentEventType.TOOL_EXECUTION_END: self._handle_tool_end,
            AgentEventType.STATS: self._handle_stats,
            AgentEventType.ERROR: self._handle_error,
            # 静默处理的生命周期事件
            AgentEventType.AGENT_START: self._handle_silent,
            AgentEventType.AGENT_END: self._handle_silent,
            AgentEventType.TURN_START: self._handle_silent,
            AgentEventType.TURN_END: self._handle_silent,
        }
    
    def render_event(self, event: AgentEvent) -> None:
        """渲染单个事件 - 简化版主入口."""
        # 类型切换时添加分隔符
        if self._need_separator(event.type):
            print("\n")
        self._prev_type = event.type

        # 使用映射表分发处理
        handler = self._event_handlers.get(event.type)
        if handler:
            handler(event)
        else:
            log(f"Unknown event type: {event.type}")
    
    def _handle_message_start(self, event: AgentEvent) -> None:
        """处理消息开始事件."""
        if self._indicator:
            self._indicator.stop()
        self._indicator = StreamingIndicator()
        self._indicator.start()
        self._reset_render_state()
    
    def _handle_reasoning_token(self, event: AgentEvent) -> None:
        """处理推理 Token."""
        if self._indicator:
            self._indicator.on_token()
        self._render_reasoning(event.data)
    
    def _handle_content_token(self, event: AgentEvent) -> None:
        """处理内容 Token."""
        if self._indicator:
            self._indicator.on_token()
        self._render_content(event.data)
    
    def _handle_message_end(self, event: AgentEvent) -> None:
        """处理消息结束."""
        if self._indicator:
            self._indicator.stop()
            self._indicator = None
        self._finalize_output()
        self._reset_render_state()
    
    def _handle_silent(self, event: AgentEvent) -> None:
        """静默处理生命周期事件."""
        pass
    
    def _render_reasoning(self, content: str) -> None:
        """渲染推理内容."""
        if not self._r_started:
            print(f"{C.GRAY}[Reasoning]: {C.RESET}", end="", flush=True)
            self._r_started = True
        print(f"{C.GRAY}{content}{C.RESET}", end="", flush=True)
    
    def _render_content(self, content: str) -> None:
        """渲染 Assistant 内容."""
        if not self._c_started:
            # 如果有推理输出，先换行
            if self._r_started:
                print()
            print(f"[Assistant]: ", end="", flush=True)
            self._c_started = True
        print(content, end="", flush=True)
```

```python
class Shell:
    def __init__(self, cfg: Config):
        self.cfg = cfg
        self._init_components()
        self._init_commands()
    
    def _init_components(self) -> None:
        """初始化核心组件."""

        tool_registry = initialize_tools()
        
        self.daemon = self._create_daemon(has_provider, tool_registry)
        self.session_manager = self._create_session_manager()
        self.config_manager = ConfigManager()
        self.frontend = Frontend(self.daemon, self.session_manager)
    
    def _create_daemon(self, has_provider: bool, tools) -> Daemon:
        """创建 Daemon 实例."""
        if has_provider and ProviderRegistry.get("openai"):
            return Daemon(
                provider_api="openai",
                model=self.cfg.model,
                history_limit=self.cfg.history,
                tool_registry=tools,
```

```python
# tests/pty_helper.py
"""PTY 测试辅助类."""

import pty
import os
import select
import subprocess
import time
import fcntl
import signal
from typing import Optional, List
from contextlib import contextmanager


class PTYSession:
    """PTY 会话管理器."""
    
    def __init__(self, timeout: float = 10.0):
        self.timeout = timeout
        self.master_fd: Optional[int] = None
        self.slave_fd: Optional[int] = None
        self.process: Optional[subprocess.Popen] = None
        
    def start(self, command: str, cwd: Optional[str] = None, env: Optional[dict] = None):
        """启动 PTY 会话."""
        self.master_fd, self.slave_fd = pty.openpty()
        
        # 设置非阻塞模式
        fl = fcntl.fcntl(self.master_fd, fcntl.F_GETFL)
        fcntl.fcntl(self.master_fd, fcntl.F_SETFL, fl | os.O_NONBLOCK)
        
        run_env = os.environ.copy()
        if env:
            run_env.update(env)
        
        self.process = subprocess.Popen(
            command,
            shell=True,
            stdin=self.slave_fd,
            stdout=self.slave_fd,
            stderr=self.slave_fd,
            cwd=cwd,
            env=run_env,
            preexec_fn=os.setsid
        )
        
        os.close(self.slave_fd)
        self.slave_fd = None
        time.sleep(0.5)
        
    def send(self, data: str):
        """发送数据."""
        if self.master_fd is not None:
            os.write(self.master_fd, data.encode())
            time.sleep(0.1)
    
    def send_ctrl_c(self):
        """发送 Ctrl-C."""
        if self.master_fd is not None:
            os.write(self.master_fd, b'\x03')
            time.sleep(0.1)
    
    def read_available(self, timeout: float = 0.5) -> str:
        """读取可用数据."""
        if self.master_fd is None:
            return ""
        
        output = ""
        start = time.time()
        while time.time() - start < timeout:
            readable, _, _ = select.select([self.master_fd], [], [], 0.1)
            if self.master_fd in readable:
                try:
                    data = os.read(self.master_fd, 4096).decode()
                    output += data
                except (OSError, IOError):
                    break
            elif output:
                break
        return output
    
    def read_until(self, marker: str = "", timeout: Optional[float] = None) -> str:
        """读取直到看到标记."""
        timeout = timeout or self.timeout
        start = time.time()
        collected = ""
        
        while time.time() - start < timeout:
            chunk = self.read_available(timeout=0.5)
            collected += chunk
            if marker and marker in collected:
                return collected
            if self.process and self.process.poll() is not None:
                break
        return collected
    
    def is_terminated(self) -> bool:
        """检查是否已终止."""
        return self.process is not None and self.process.poll() is not None
    
    def wait_for_exit(self, timeout: float = 5.0) -> bool:
        """等待进程退出."""
        if not self.process:
            return True
        try:
            self.process.wait(timeout=timeout)
            return True
        except subprocess.TimeoutExpired:
            return False
    
    def stop(self):
        """停止会话."""
        if self.process and self.process.poll() is None:
            self.process.terminate()
            try:
                self.process.wait(timeout=2)
            except subprocess.TimeoutExpired:
                self.process.kill()
                self.process.wait()
        
        if self.master_fd is not None:
            os.close(self.master_fd)
            self.master_fd = None


class PTYTestBase:
    """PTY 测试基类."""
    
    @contextmanager
    def session(self, timeout: float = 10.0):
        """创建会话上下文."""
        s = PTYSession(timeout)
        try:
            project_root = Path(__file__).parent.parent

                cwd=project_root,
                env={"PYTHONPATH": str(project_root / "src")}
            )
            yield s
        finally:
            s.stop()
```

### 2.4 统一错误处理设计

#### 2.4.1 错误处理器设计

```python
# src/tui_chatbot/error_handler.py (已存在，需要增强)
"""统一错误处理器."""

from typing import Dict, Type, Callable, Optional
from .core.events import AgentEvent, AgentEventType


class ErrorHandler:
    """统一错误处理器."""
    
    # 错误消息模板
    ERROR_MESSAGES: Dict[Type[Exception], str] = {
        ValueError: "输入错误: {message}",
        TypeError: "类型错误: {message}",
        RuntimeError: "运行时错误: {message}",
        ConnectionError: "连接失败: {message}",
        TimeoutError: "操作超时，请重试",
        KeyboardInterrupt: "操作已取消",
        asyncio.CancelledError: "操作已中断",
    }
    
    @classmethod
    def handle(
        cls, 
        error: Exception, 
        context: str = "",
        fallback: str = "发生未知错误"
    ) -> str:
        """处理错误并返回用户友好的消息.
        
        Args:
            error: 异常对象
            context: 错误上下文（如"保存会话"）
            fallback: 默认错误消息
            
        Returns:
            格式化后的错误消息
        """
        error_type = type(error)
        template = cls.ERROR_MESSAGES.get(error_type, fallback + ": {message}")
        
        message = template.format(message=str(error))
        if context:
            message = f"[{context}] {message}"
        
        return message
    
    @classmethod
    def create_event(cls, error: Exception, context: str = "") -> AgentEvent:
        """创建错误事件.
        
        Args:
            error: 异常对象
            context: 错误上下文
            
        Returns:
            AgentEvent 错误事件
        """
        message = cls.handle(error, context)
        return AgentEvent(
            type=AgentEventType.ERROR,
            error=message,
            original_error=error
        )
```
