#### 2.1.3 TokenStats.elapsed 计算修复 (B3)

**当前代码**:
```python
@property
def elapsed(self) -> float:
    """总耗时 (秒)."""
    if self.start_time is None:
        return 0.0
    end = self.end_time or time.time()
    return max(0.0, end - self.start_time)
```

**问题分析**:
- 当 `start_time=100.0` 且 `end_time=None` 时，使用 `time.time()` 作为结束时间
- 但测试期望返回 `0.0`，表示未完成的统计不应计算耗时

**目标代码**:
```python
@property
def elapsed(self) -> float:
    """总耗时 (秒)."""
    if self.start_time is None or self.end_time is None:
        return 0.0
    return max(0.0, self.end_time - self.start_time)
```

**影响分析**:
- 修改 `src/tui_chatbot/core/events.py` 第 212-218 行
- 测试 `test_elapsed_zero_when_no_end` 将通过
- 语义更清晰：只有完成的统计才返回非零耗时

#### 2.2.3 会话管理器设计 (manager.py)

```python
from typing import Dict, List, Optional, Callable
import uuid

from .models import ChatSession, SessionMetadata
from .storage import SessionStorage


class SessionManager:
    """会话管理器 - 管理多个聊天会话的生命周期."""
    
    def __init__(
        self,
        storage: Optional[SessionStorage] = None,
        on_session_change: Optional[Callable[[str], None]] = None,
    ):
        self._sessions: Dict[str, ChatSession] = {}
        self._current_id: Optional[str] = None
        self._storage = storage or SessionStorage()
        self._on_change = on_session_change
    
    # ═══════════════════════════════════════════════════════════════
    # 会话 CRUD 操作
    # ═══════════════════════════════════════════════════════════════
    
    def create_session(
        self,
        title: Optional[str] = None,
        model: str = "gpt-4o",
    ) -> ChatSession:
        """创建新会话."""
        session_id = str(uuid.uuid4())[:8]  # 短 ID 便于使用
        metadata = SessionMetadata(
            id=session_id,
            title=title or f"Chat {len(self._sessions) + 1}",
            model=model,
        )
        session = ChatSession(metadata=metadata)
        self._sessions[session_id] = session
        self._current_id = session_id
        
        # 异步保存
        self._storage.save_async(session)
        
        if self._on_change:
            self._on_change(session_id)
        
        return session
    
    def get_session(self, session_id: str) -> Optional[ChatSession]:
        """获取指定会话."""
        return self._sessions.get(session_id)
    
    def get_current_session(self) -> Optional[ChatSession]:
        """获取当前会话."""
        if self._current_id:
            return self._sessions.get(self._current_id)
        return None
    
    def switch_session(self, session_id: str) -> bool:
        """切换到指定会话."""
        if session_id in self._sessions:
            self._current_id = session_id
            if self._on_change:
                self._on_change(session_id)
            return True
        return False
    
    def delete_session(self, session_id: str) -> bool:
        """删除会话."""
        if session_id in self._sessions:
            del self._sessions[session_id]
            self._storage.delete(session_id)
            
            # 如果删除的是当前会话，切换到其他会话
            if self._current_id == session_id:
                self._current_id = next(iter(self._sessions.keys()), None)
                if self._on_change:
                    self._on_change(self._current_id)
            
            return True
        return False
    
    def list_sessions(self) -> List[SessionMetadata]:
        """列出所有会话元数据."""
        return [s.metadata for s in self._sessions.values()]
    
    def rename_session(self, session_id: str, new_title: str) -> bool:
        """重命名会话."""
        session = self._sessions.get(session_id)
        if session:
            session.metadata.title = new_title
            session.metadata.updated_at = datetime.now()
            self._storage.save_async(session)
            return True
        return False
    
    # ═══════════════════════════════════════════════════════════════
    # 持久化操作
    # ═══════════════════════════════════════════════════════════════
    
    async def load_all(self) -> None:
        """从存储加载所有会话."""
        sessions = await self._storage.load_all()
        self._sessions = {s.metadata.id: s for s in sessions}
        
        # 默认选择最新的会话
        if self._sessions:
            latest = max(
                self._sessions.values(),
                key=lambda s: s.metadata.updated_at,
            )
            self._current_id = latest.metadata.id
    
    async def save_all(self) -> None:
        """保存所有会话到存储."""
        for session in self._sessions.values():
            await self._storage.save(session)
```

#### 2.2.4 存储层设计 (storage.py)

```python
import json
import os
import aiofiles
from pathlib import Path
from typing import List, Optional
from datetime import datetime

from .models import ChatSession, SessionMetadata


class SessionStorage:
    """会话存储 - 基于 JSON 文件的持久化."""
    
    DEFAULT_DIR = Path.home() / ".tui_chatbot" / "sessions"
    
    def __init__(self, storage_dir: Optional[Path] = None):
        self._dir = storage_dir or self.DEFAULT_DIR
        self._ensure_dir()
    
    def _ensure_dir(self) -> None:
        """确保存储目录存在."""
        self._dir.mkdir(parents=True, exist_ok=True)
    
    def _get_file_path(self, session_id: str) -> Path:
        """获取会话文件路径."""
        return self._dir / f"{session_id}.json"
    
    async def save(self, session: ChatSession) -> None:
        """保存会话到文件."""
        file_path = self._get_file_path(session.metadata.id)
        data = session.to_dict()
        
        async with aiofiles.open(file_path, 'w', encoding='utf-8') as f:
            await f.write(json.dumps(data, indent=2, default=str))
    
    def save_async(self, session: ChatSession) -> None:
        """异步保存（后台执行）."""
        import asyncio
        asyncio.create_task(self.save(session))
    
    async def load(self, session_id: str) -> Optional[ChatSession]:
        """加载指定会话."""
        file_path = self._get_file_path(session_id)
        
        if not file_path.exists():
            return None
        
        async with aiofiles.open(file_path, 'r', encoding='utf-8') as f:
            content = await f.read()
            data = json.loads(content)
            return ChatSession.from_dict(data)
    
    async def load_all(self) -> List[ChatSession]:
        """加载所有会话."""
        sessions = []
        
        if not self._dir.exists():
            return sessions
        
        for file_path in self._dir.glob("*.json"):
            session_id = file_path.stem
            session = await self.load(session_id)
            if session:
                sessions.append(session)
        
        # 按更新时间排序
        sessions.sort(key=lambda s: s.metadata.updated_at, reverse=True)
        return sessions
    
    def delete(self, session_id: str) -> bool:
        """删除会话文件."""
        file_path = self._get_file_path(session_id)
        
        if file_path.exists():
            file_path.unlink()
            return True
        return False
    
    async def export_session(self, session_id: str, export_path: Path) -> bool:
        """导出会话到指定路径."""
        session = await self.load(session_id)
        if not session:
            return False
        
        data = session.to_dict()
        async with aiofiles.open(export_path, 'w', encoding='utf-8') as f:
            await f.write(json.dumps(data, indent=2, default=str))
        
        return True
```

```python
# src/tui_chatbot/config/manager.py

import json
import os
from pathlib import Path
from typing import Optional, Dict, Any
from dataclasses import dataclass, asdict
import aiofiles


@dataclass
class UserConfig:
    """用户配置."""
    
    # 默认模型设置
    default_model: str = "gpt-4o"
    default_temperature: float = 0.7
    default_max_tokens: Optional[int] = None
    
    # UI 设置
    theme: str = "dark"
    font_size: int = 14
    
    # API 设置
    openai_api_key: Optional[str] = None
    anthropic_api_key: Optional[str] = None
    ollama_base_url: str = "http://localhost:11434"
    
    # 行为设置
    auto_save: bool = True
    save_history: bool = True
    max_history_length: int = 1000


class ConfigManager:
    """配置管理器 - 管理用户配置的持久化."""
    
    CONFIG_FILE = Path.home() / ".tui_chatbot" / "config.json"
    
    def __init__(self):
        self._config = UserConfig()
        self._loaded = False
    
    async def load(self) -> UserConfig:
        """从文件加载配置."""
        if self._loaded:
            return self._config
        
        if self.CONFIG_FILE.exists():
            try:
                async with aiofiles.open(self.CONFIG_FILE, 'r') as f:
                    content = await f.read()
                    data = json.loads(content)
                    
                    # 只更新存在的字段
                    for key, value in data.items():
                        if hasattr(self._config, key):
                            setattr(self._config, key, value)
                
                self._loaded = True
            except Exception:
                # 加载失败使用默认配置
                pass
        
        return self._config
    
    async def save(self) -> None:
        """保存配置到文件."""
        self.CONFIG_FILE.parent.mkdir(parents=True, exist_ok=True)
        
        async with aiofiles.open(self.CONFIG_FILE, 'w') as f:
            await f.write(json.dumps(asdict(self._config), indent=2))
    
    @property
    def config(self) -> UserConfig:
        """获取当前配置."""
        return self._config
    
    def update(self, **kwargs) -> None:
        """更新配置."""
        for key, value in kwargs.items():
            if hasattr(self._config, key):
                setattr(self._config, key, value)
```
